Conceitos Básicos

📦 O que são Objetos?

Objetos são estruturas que armazenam dados relacionados em pares chave-valor. Como uma caixa organizada com etiquetas!

// Um objeto é como uma caixa com compartimentos const pessoa = { nome: "Ana", // 🧱 propriedade idade: 25, // 🧱 propriedade falar: function() { // 🧰 método return "Olá!"; } };
Dica: Pense em objetos como fichas de personagens de RPG - cada um tem suas características únicas!
🧱 Propriedades

Propriedades são as características do objeto. Como os atributos de um personagem!

const carro = { marca: "Toyota", // 🧱 propriedade cor: "azul", // 🧱 propriedade ano: 2023 // 🧱 propriedade }; // Acessando propriedades console.log(carro.marca); // "Toyota" console.log(carro["cor"]); // "azul"
🧰 Métodos

Métodos são funções dentro de objetos. São as "habilidades" que o objeto pode executar!

const calculadora = { somar: function(a, b) { // 🧰 método return a + b; }, multiplicar(a, b) { // 🧰 método (sintaxe moderna) return a * b; } }; console.log(calculadora.somar(2, 3)); // 5

Nível Iniciante

➕ Adicionando Propriedades
const jogador = { nome: "Mario" }; // Adicionando novas propriedades jogador.pontos = 100; jogador["vidas"] = 3; jogador.poder = "Super"; console.log(jogador); // { nome: "Mario", pontos: 100, vidas: 3, poder: "Super" }
🗑️ Removendo Propriedades
const inventario = { espada: 1, escudo: 1, pocao: 5, lixo: "item inútil" }; // Removendo propriedade delete inventario.lixo; console.log(inventario); // { espada: 1, escudo: 1, pocao: 5 }
🔄 Loop for...in
const pokemon = { nome: "Pikachu", tipo: "Elétrico", nivel: 25, hp: 100 }; // Percorrendo todas as propriedades for (let propriedade in pokemon) { console.log(`${propriedade}: ${pokemon[propriedade]}`); } // nome: Pikachu // tipo: Elétrico // nivel: 25 // hp: 100
🔍 Verificando Propriedades
const usuario = { nome: "João", email: "joao@email.com" }; // Verificando se propriedade existe console.log("nome" in usuario); // true console.log("telefone" in usuario); // false // Usando hasOwnProperty console.log(usuario.hasOwnProperty("email")); // true

Nível Intermediário

🔑 Object.keys()
const loja = { notebook: 2500, mouse: 50, teclado: 150, monitor: 800 }; // Obtendo todas as chaves const produtos = Object.keys(loja); console.log(produtos); // ["notebook", "mouse", "teclado", "monitor"] // Contando produtos console.log(`Temos ${produtos.length} produtos`);
💎 Object.values()
const notas = { matematica: 8.5, portugues: 9.0, historia: 7.5, ciencias: 8.0 }; // Obtendo todos os valores const todasNotas = Object.values(notas); console.log(todasNotas); // [8.5, 9.0, 7.5, 8.0] // Calculando média const media = todasNotas.reduce((a, b) => a + b) / todasNotas.length; console.log(`Média: ${media}`); // Média: 8.25
🔗 Object.entries()
const ranking = { primeiro: "Ana", segundo: "Bruno", terceiro: "Carlos" }; // Obtendo pares chave-valor const posicoes = Object.entries(ranking); console.log(posicoes); // [["primeiro", "Ana"], ["segundo", "Bruno"], ["terceiro", "Carlos"]] // Usando com forEach posicoes.forEach(([posicao, nome]) => { console.log(`${posicao}: ${nome}`); });
🎯 Destructuring
const heroi = { nome: "Superman", poder: "Voo", cidade: "Metrópolis", identidade: "Clark Kent" }; // Extraindo propriedades const { nome, poder, cidade } = heroi; console.log(nome); // "Superman" console.log(poder); // "Voo" console.log(cidade); // "Metrópolis" // Com renomeação const { identidade: nomeReal } = heroi; console.log(nomeReal); // "Clark Kent"

Nível Avançado

🏗️ Função Construtora
// Função construtora (como um molde) function Carro(marca, modelo, ano) { this.marca = marca; this.modelo = modelo; this.ano = ano; this.acelerar = function() { return `${this.marca} ${this.modelo} acelerando! 🚗💨`; }; } // Criando instâncias const meuCarro = new Carro("Honda", "Civic", 2023); const outroCarro = new Carro("Toyota", "Corolla", 2022); console.log(meuCarro.acelerar());
🧬 Prototype
// Adicionando método ao prototype Carro.prototype.frear = function() { return `${this.marca} ${this.modelo} freando! 🛑`; }; // Todos os carros agora têm o método frear console.log(meuCarro.frear()); console.log(outroCarro.frear()); // Verificando prototype console.log(meuCarro.__proto__ === Carro.prototype); // true
👑 Classes (ES6)
class Animal { constructor(nome, tipo) { this.nome = nome; this.tipo = tipo; } falar() { return `${this.nome} faz algum som! 🔊`; } } class Cachorro extends Animal { constructor(nome, raca) { super(nome, "Cachorro"); this.raca = raca; } falar() { return `${this.nome} faz: Au au! 🐕`; } } const rex = new Cachorro("Rex", "Labrador"); console.log(rex.falar());
🔒 Getters e Setters
class ContaBancaria { constructor(titular) { this.titular = titular; this._saldo = 0; // propriedade "privada" } get saldo() { return `R$ ${this._saldo.toFixed(2)}`; } set saldo(valor) { if (valor >= 0) { this._saldo = valor; } else { console.log("Saldo não pode ser negativo!"); } } depositar(valor) { this._saldo += valor; return `Depósito realizado! Saldo: ${this.saldo}`; } } const conta = new ContaBancaria("Maria"); console.log(conta.depositar(100));
